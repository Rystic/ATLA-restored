harass_tactic = {
	days = 18
	sprite = 4
	group = harass

	trigger = {
		phase = skirmish
		OR = {
			light_cavalry = 0.05
		}
		OR = {
			NOT = {
				OR = {
					light_cavalry = 0.45
				}
			}
			AND = {
				flank_has_leader = yes
				leader = { martial = 12 }
				NOT = {
					OR = {
						light_cavalry = 0.75
					}
				}
			}
		}
	}

	mean_time_to_happen = {
		days = 2
		modifier = {
			factor = 2
			flank_has_leader = yes
			leader = {
				martial = 8
			}
			OR = {
				light_troops = {
					who = light_cavalry
					value = 0.3
				}
			}
		}
		modifier = {
			factor = 2
			flank_has_leader = yes
			leader = {
				martial = 12
			}
			OR = {
				light_troops = {
					who = light_cavalry
					value = 0.4
				}
			}
		}

			factor = 2

			}
			OR = {
