#Fire Sub-Bending
can_teach_sub_bending_lightning = {
	OR = {
		trait = firebender
		trait = avatar_firebender
	}
	trait = lightningbender
	FROM = {
		OR = {
			trait = firebender
			trait = avatar_firebender
		}
		NOT = {
			trait = lightningbender
		}
	}
}

can_teach_sub_bending_seismicsense = {
	OR = {
		trait = earthbender
		trait = avatar_earthbender
	}
	OR = {
		trait = blind
		trait = blinded
	}
	trait = seismicsense
	FROM = {
		OR = {
			trait = earthbender
			trait = avatar_earthbender
		}
		OR = {
			trait = blind
			trait = blinded
		}
		NOR = {
			trait = seismicsense
			trait = seismicsense_nonblind
		}
	}
}
