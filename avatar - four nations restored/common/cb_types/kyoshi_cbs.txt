war_for_hearts_and_minds = {
	name = CB_HEARTS_AND_MINDS
	war_name = WAR_HEARTS_AND_MINDS
	sprite = 9
	truce_days = 1325
	hostile_against_others = yes
	is_permanent = yes
	check_all_titles = yes # if permanent, setting this to true will check against all of someones titles, including vassal held titles
	can_ask_to_join_war = no
	is_holy_war = yes
	allow_distant = yes
	allowed_to_target_tributaries = yes

	can_use = {
		ROOT = {
			mercenary = no
			religion = followers_of_kyoshi
		}
		FROM = {
			mercenary = no
			religion = way_of_chin
		}
	}

	can_use_title = {
		tier = count
		location = {
			any_neighbor_province = {
					AND = {
						has_owner = yes
						owner = {
							OR = {
								character = ROOT
								is_liege_or_above = ROOT
							}
						}
					}
					AND = {
						is_land = no
						any_neighbor_province = {
							OR = {
								AND = {
									has_owner = yes
									owner = {
										OR = {
											character = ROOT
											is_liege_or_above = ROOT
										}
									}
								}
								AND = {
									is_land = no
									any_neighbor_province = {
										owner = {
											OR = {
												character = ROOT
												is_liege_or_above = ROOT
											}
										}
									}
								}
							}
						}
					}
				}
			}
		}

	is_valid = {
		ROOT = {
			religion = followers_of_kyoshi
		}
	}

	on_success_title = {
		add_claim = ROOT

		prestige = 100

		ROOT = {
			religion_authority = {
				modifier = won_holy_war
				years = 20
			}
		}

		FROM = {
			religion_authority = {
				modifier = lost_holy_war
				years = 20
			}
		}

		custom_tooltip = {
			text = hearts_and_minds_lose_propaganda
			hidden_tooltip = {
				FROM = {
					subtract_variable = { which = propaganda value = 75 }
				}
			}
		}
	}

	on_fail_title = {
		prestige = -100

		custom_tooltip = {
			text = hearts_and_minds_gain_propaganda_small
			hidden_tooltip = {
				FROM = {
					change_variable = { which = propaganda value = 25 }
				}
			}
		}
	}

	on_reverse_demand = {

		prestige = -300
		ROOT = {
			religion_authority = {
				modifier = lost_holy_war
				years = 20
			}
		}

		FROM = {

			prestige = 200

			religion_authority = {
				modifier = won_holy_war
				years = 20
			}
		}

		custom_tooltip = {
			text = hearts_and_minds_gain_propaganda_large
			hidden_tooltip = {
				FROM = {
					change_variable = { which = propaganda value = 75 }
				}
			}
		}
	}

	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}

	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}

	defender_ai_defeat_worth = {
		factor = 100
	}

	ai_will_do = {
		factor = 1000
	}
}
