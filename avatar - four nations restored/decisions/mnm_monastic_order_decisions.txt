	monastic_orders_donate_to_charity = {
		only_playable = yes
		ai_check_interval = 12

		potential = {
			society_member_of = monastic_order_elder_council
			OR = {
				has_quest = quest_monastic_order_donate_to_charity # you don't have to wait a year-ish to donate money to complete a quest
				NOT = { has_character_modifier = recently_donated_to_charity}
			}
		}

		allow = {
			wealth = 100
		}

		effect = {
			tooltip = {
				wealth = -100
				if = {
					limit = { has_quest = quest_monastic_order_donate_to_charity }
					add_society_currency_medium_effect = yes
				}
				if = {
					limit = { NOT = { has_quest = quest_monastic_order_donate_to_charity } }
					add_society_currency_minor_effect = yes
				}
				add_character_modifier = {
					modifier = recently_donated_to_charity
					days = 360
				}
				if = {
					limit = {
						NOR = {
							trait = charitable
							trait = greedy
						}
					}
					random = {
						chance = 30
						add_trait = charitable
					}
				}
				if = {
					limit = { trait = greedy }
					random = {
						chance = 40
						remove_trait = greedy
					}
				}
			}
			hidden_effect = {
				add_character_modifier = {
					modifier = recently_donated_to_charity
					days = 360
				}
				wealth = -100

				#searches for a random character society member to give the money to
				random_society_member = {
					limit = {
						same_society_as = ROOT
						is_adult = yes
						NOT = { has_job_action = yes }
						ai = yes
						NOT = { character = ROOT } #don't want to donate to ourselves
						#NOT = { society_rank = 3 }
						is_society_grandmaster = no #head dude shouldnt get the money
					}
					wealth = 100
					#set_character_flag = got_Some_Money #This flag is for finding out who got the money
				}

				if = {
					limit = { NOT = { has_quest = quest_monastic_order_donate_to_charity } }
					add_society_currency_minor_effect = yes
				}

				if = {
					limit = { has_quest = quest_monastic_order_donate_to_charity }
					clr_quest = quest_monastic_order_donate_to_charity
					sound_effect = monastic_orders_donate_charity
					hidden_effect = { monastic_order_purity_effect = yes }
					add_society_currency_medium_effect = yes
					character_event = { id = MNSAK.3007 } #lets the leader of the society tell you that you completed the quest to donate to charity
				}

				if = {
					limit = { NOT = { trait = greedy } }
					random = {
						chance = 30
						add_trait = charitable
					}
				}
				if = {
					limit = { trait = greedy }
					random = {
						chance = 40
						remove_trait = greedy
					}
				}
			}
		}
	}
