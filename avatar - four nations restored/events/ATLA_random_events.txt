		text = EVTDESC_CASE2_ATLAR.764

			AND = {

				d_avatar = {
					has_holder = yes
				}
			}
		}

	immediate = {
		if = {
			limit = {
				d_avatar = {
					has_holder = yes
				}

		name = EVTOPTA_ATLAR.764
