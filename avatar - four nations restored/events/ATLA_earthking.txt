			title = e_earth_kingdom
		}

			trait = earthkingpuppet
			trait = earthking
			trait = earthkingconstitution
		}
	}
	option = {
		name = EVTOPTearthking.1
		if = {
			limit = {
				NOT = { check_variable = { which = global_royalAuthority value >= -100 } }
			}
			set_variable = { which = global_royalAuthority value = 50 }
		}
		if = {
			limit = {
				NOR = { 
					has_global_flag = earthking_in_power
					has_global_flag = kyoshi_parliamental_monarch
				}
			}
			add_trait = earthkingpuppet
			add_character_modifier = { name = supporting_the_earth_kingdom duration = -1 }
			clr_global_flag = restoration_ambition_delay
			subtract_variable = { which = global_royalAuthority value = 30 }
		}
		if = {
			limit = { 
				OR = {
					has_global_flag = kyoshi_parliamental_monarch
					has_global_flag = ek_gave_some_power 
					
				}
			}
			add_trait = earthkingconstitution
			add_character_modifier = { name = supporting_the_earth_kingdom duration = -1 }
		}
		if = {
			limit = { 
				has_global_flag = earthking_in_power
			}
			add_trait = earthking
			add_character_modifier = { name = supporting_the_earth_kingdom duration = -1 }
		}
		character_event = { id = ATLAE.211 }
	}
}

character_event = {
	id = earthking.4
	picture = GFX_evt_landscape_3
	border = GFX_event_normal_frame_diplomacy
	desc = EVTDESCEK.4

	only_rulers = yes
	capable_only = yes
	prisoner = no
	only_independent = yes

	is_triggered_only = yes

	trigger = {
		mercenary = no
		is_tributary = yes
		NOT = { 
			trait = in_hiding
		}
		any_allied_character = {
			OR = {
				trait = earthkingpuppet
				trait = earthking
			}
		}
	}

	immediate = {
		ROOT = { capital_scope = { save_event_target_as = capital } }
		if = {
			limit = {
				any_allied_character = {
					OR = {
						trait = earthkingpuppet
						trait = earthking
					}
				}
			}
			random_allied_character = {
				limit = {
					OR = {
						trait = earthkingpuppet
						trait = earthking
					}
				}
				save_event_target_as = target_ruler
				capital_scope = { save_event_target_as = target_ruler_capital }
			}
		}
	}
	#write a formal complaint to King
	option = {
		name = EVTOPTA_EK_4
		ai_chance = { 
			factor = 90
		}
		custom_tooltip = {
			text = EVTOPTA_EK_4_TOOLTIP
			hidden_tooltip = {
				event_target:target_ruler = {
					character_event = { id = earthking.5 days = 3 }
				}
			}
		}
		prestige = 50
	}
	# Ignore it
	option = {
		name = EVTOPTB_EK_4
		ai_chance = { 
			factor = 10
		}
		custom_tooltip = {
			text = EVTOPTB_EK_4_TOOLTIP
			hidden_tooltip = {
				any_courtier = {
					opinion = {
						modifier = opinion_disappointed
						who = ROOT
						months = 24
					}
				}
			}
		}
		scaled_wealth = -0.15
	}
	# We will take care of it ourselves
	option = {
		trigger = {
			intrigue = 15
		}
		name = EVTOPTC_EK_4
		custom_tooltip = {
			text = EVTOPTC_EK_4_TOOLTIP
		}
		prestige = 50
		tooltip_info = intrigue
	}
}
