			limit = {
				has_dynasty_flag = ek_tyranny
			}
			add_character_modifier = { name = ek_tyranny duration = -1 }
			remove_character_modifier = ek_good_authority
		}
		if = {
			limit = {
				NOT = { has_global_flag = earthking_in_power }
			}
			add_trait = earthkingpuppet
			add_character_modifier = { name = supporting_the_earth_kingdom duration = -1 }
			clr_global_flag = restoration_ambition_delay
			subtract_variable = { which = global_royalAuthority value = 30 }
		}
		
		if = { # Calculate Royal Court Power
			limit = {
				has_global_flag = earthking_in_power
			}
			add_trait = earthking
			add_character_modifier = { name = supporting_the_earth_kingdom duration = -1 }
			subtract_variable = { which = global_royalAuthority value = 15 }
			if = {
				limit = {
					NOT = { age = 12 }
					is_title_active = d_dai_li
				}
				character_event = { id = earthking.3 }
			}
		}
		character_event = { id = ATLAE.211 }
	}
}
