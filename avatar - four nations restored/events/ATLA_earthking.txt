namespace = earthking

character_event = {
	id = earthking.1
	is_triggered_only = yes
	hide_window = yes
	only_rulers = yes

	trigger = {
		OR = {
			has_landed_title = e_earth_kingdom
			has_landed_title = k_earth_kingdom
		}
		independent = yes
		NOR = {
			trait = earthkingpuppet
			trait = earthking
			trait = earthkingconstitution
		}
	}
	option = {
		if = {
			limit = {
				NOT = { check_variable = { which = global_royalAuthority value >= -100 } }
			}
			set_variable = { which = global_royalAuthority value = 50 }
		}
		if = {
			limit = {
				NOR = {
					has_global_flag = earthking_in_power
					has_global_flag = kyoshi_parliamental_monarch
				}
			}
			add_trait = earthkingpuppet
			add_character_modifier = { name = supporting_the_earth_kingdom duration = -1 }
			clr_global_flag = restoration_ambition_delay
			subtract_variable = { which = global_royalAuthority value = 30 }
		}
		if = {
			limit = {
				OR = {
					has_global_flag = kyoshi_parliamental_monarch
					has_global_flag = ek_gave_some_power

				}
			}
			add_trait = earthkingconstitution
			add_character_modifier = { name = supporting_the_earth_kingdom duration = -1 }
		}
		if = {
			limit = {
				has_global_flag = earthking_in_power
			}
			add_trait = earthking
			add_character_modifier = { name = supporting_the_earth_kingdom duration = -1 }
		}
		character_event = { id = ATLAE.211 }

		if = {
			limit = {
				AND = {
					k_earth_kingdom = {
						has_holder = yes
					}
					e_earth_kingdom = {
						has_holder = no
					}
				}
			}
			k_earth_kingdom = {
				save_persistent_event_target = {
					name = Current_EK_Title
					scope = THIS
				}
			}
		}
		else_if = {
			limit = {
				AND = {
					k_earth_kingdom = {
						has_holder = no
					}
					e_earth_kingdom = {
						has_holder = yes
					}
				}
			}
			e_earth_kingdom = {
				save_persistent_event_target = {
					name = Current_EK_Title
					scope = THIS
				}
			}
		}
	}
}

# Noble complains of Banditry
character_event = {
	id = earthking.4
	picture = GFX_evt_landscape_3
	border = GFX_event_normal_frame_diplomacy
	desc = EVTDESCEK.4

	only_rulers = yes
	capable_only = yes
	prisoner = no
	only_independent = yes

	is_triggered_only = yes

	trigger = {
		mercenary = no
		is_tributary = yes
		NOT = {
			trait = in_hiding
		}
		any_allied_character = {
			OR = {
				trait = earthkingpuppet
				trait = earthking
			}
		}
	}

	immediate = {
		ROOT = { capital_scope = { save_event_target_as = capital } }
		if = {
			limit = {
				any_allied_character = {
					OR = {
						trait = earthkingpuppet
						trait = earthking
					}
				}
			}
			random_allied_character = {
				limit = {
					OR = {
						trait = earthkingpuppet
						trait = earthking
					}
				}
				save_event_target_as = target_ruler
				capital_scope = { save_event_target_as = target_ruler_capital }
			}
		}
	}
	#write a formal complaint to King
	option = {
		name = EVTOPTA_EK_4
		ai_chance = {
			factor = 90
		}
		custom_tooltip = {
			text = EVTOPTA_EK_4_TOOLTIP
			hidden_tooltip = {
				event_target:target_ruler = {
					character_event = { id = earthking.5 days = 3 }
				}
			}
		}
		prestige = 50
	}
	# Ignore it
	option = {
		name = EVTOPTB_EK_4
		ai_chance = {
			factor = 10
		}
		custom_tooltip = {
			text = EVTOPTB_EK_4_TOOLTIP
			hidden_tooltip = {
				any_courtier = {
					opinion = {
						modifier = opinion_disappointed
						who = ROOT
						months = 24
					}
				}
			}
		}
		scaled_wealth = -0.15
	}
	# We will take care of it ourselves
	option = {
		trigger = {
			intrigue = 15
		}
		name = EVTOPTC_EK_4
		custom_tooltip = {
			text = EVTOPTC_EK_4_TOOLTIP
		}
		prestige = 50
		tooltip_info = intrigue
	}
}

# Noble complains of Raiders
character_event = {
	id = earthking.18
	picture = GFX_evt_chin_army
	border = GFX_event_normal_frame_diplomacy
	desc = EVTDESCEK.18

	only_rulers = yes
	capable_only = yes
	prisoner = no
	only_independent = yes

	is_triggered_only = yes

	trigger = {
		mercenary = no
		is_tributary = yes
		NOT = { has_character_flag = complained_about_raiders }
		NOT = {
			trait = in_hiding
		}
		any_allied_character = {
			OR = {
				trait = earthkingpuppet
				trait = earthking
			}
		}
	}

	immediate = {
		set_character_flag = complained_about_raiders
		ROOT = { capital_scope = { save_event_target_as = capital } }
		if = {
			limit = {
				any_allied_character = {
					OR = {
						trait = earthkingpuppet
						trait = earthking
					}
				}
			}
			random_allied_character = {
				limit = {
					OR = {
						trait = earthkingpuppet
						trait = earthking
					}
				}
				save_event_target_as = target_ruler
				capital_scope = { save_event_target_as = target_ruler_capital }
			}
		}
	}
	#write a formal complaint to King
	option = {
		name = EVTOPTA_EK_4
		ai_chance = {
			factor = 90
		}
		custom_tooltip = {
			text = EVTOPTA_EK_4_TOOLTIP
			hidden_tooltip = {
				event_target:target_ruler = {
					character_event = { id = earthking.20 days = 3 }
				}
			}
		}
		prestige = 50
	}
	# Ignore it
	option = {
		name = EVTOPTB_EK_4
		ai_chance = {
			factor = 10
		}
		custom_tooltip = {
			text = EVTOPTB_EK_4_TOOLTIP
			hidden_tooltip = {
				any_courtier = {
					opinion = {
						modifier = opinion_disappointed
						who = ROOT
						months = 24
					}
				}
			}
		}
	}
}

	is_triggered_only = yes

	# Fine the Raider
	option = {
		trigger = {
			FROMFROM = {

					OR = {

		name = EVTOPTA_EK_20
		ai_chance = {
			factor = 70

		}
		custom_tooltip = {
			text = EARTHKINGDOM_AUTHORITY_MINOR_BOOST
			hidden_tooltip = {
				change_variable = { which = global_royalAuthority value = 1 }
			}
		}
		FROMFROM = {
			transfer_scaled_wealth = { to = FROM value = 0.1 }
		}
		prestige = 100
		tooltip_info = diplomacy
	}

	# Promise Aid
	option = {
		name = EVTOPTB_EK_20
		ai_chance = {
			factor = 20
		}
		custom_tooltip = {
			text = EVTOPTB_EK_20_TOOLTIP
			hidden_tooltip = {
				FROM = { letter_event = { id = earthking.22 days = 3 } }
			}
		}

		ROOT = {
			transfer_scaled_wealth = { to = FROM value = 0.05 }
		}
	}

	# Refuse to interfere
	option = {
		name = EVTOPTC_EK_20
		ai_chance = {
			factor = 10
		}
		custom_tooltip = {
			text = EVTOPTC_EK_20_TOOLTIP
			hidden_tooltip = {
				FROM = { letter_event = { id = earthking.23 days = 3 } }
			}
		}
		custom_tooltip = {
			text = EARTHKINGDOM_AUTHORITY_MINOR_LOSS
			hidden_tooltip = {
				change_variable = { which = global_royalAuthority value = -1 }
			}
		}
		prestige = 50
	}
