# Hidden start: Send the rank-up request to the current Grandmaster (delayed event on self, from decision to rank up)
character_event = {
	id = MNSAK.3000

	hide_window = yes

	trigger = {
		has_character_flag = has_sent_request_to_rank_up
	}

	is_triggered_only = yes

	immediate = {
		if = {
			limit = { is_in_society = yes }
			random_society_member = {
				limit = {
					is_society_grandmaster = yes
					same_society_as = ROOT
				}
				character_event = { id = MNSAK.3001 } # send request to Grand Master. Relic rewards also handled by this event
			}
		}
		if = {
			limit = { is_in_society = no }
			clr_character_flag = has_sent_request_to_rank_up
		}
	}
}

		if = {
			limit = { NOT = { has_quest = quest_kes_donate_to_charity } }

			limit = { has_quest = quest_kes_donate_to_charity }

			modifier = recently_donated_to_charity
			days = 360
		}
		if = {
			limit = { NOT = { trait = greedy } }
			random = {
				chance = 30
				add_trait = charitable
			}
		}
		if = {
			limit = { trait = greedy }
			random = {
				chance = 40
				remove_trait = greedy
			}
		}
		clr_character_flag = donate_to_charity_cooldown
	}
}

	immediate = {

				modifier = { factor = 0 is_playable = no }
				modifier = { factor = 0 society_rank >= 3 }
				set_character_flag = kes_quest_act_against_rulers
			}
			10 = {
				modifier = { factor = 0 NOT = { has_dlc ="Way of Life" } }

				modifier = { factor = 0 is_playable = no } # Only playable characters get focuses

			}
			10 = {
				modifier = { factor = 0 is_playable = no }
				set_character_flag = kes_quest_train
			}
			10 = {
				modifier = { factor = 0 is_playable = no }
				set_character_flag = kes_invite_member
			}
			50 = { # AI bypasses missions at times
				modifier = {
					factor = 0
					ai = no
				}
				random_list = {
					70 = {
						modifier = {
							factor = 0.9

							society_influence >= 90
						}

					}

				modifier = { factor = 0 is_playable = no }
				modifier = { factor = 0 society_rank >= 3 }
				set_character_flag = twl_quest_pacify_rebellion
			}
			10 = {
				modifier = { factor = 0 NOT = { has_dlc ="Way of Life" } }
				modifier = { factor = 0 has_focus = focus_intrigue }
				modifier = { factor = 0 is_playable = no } # Only playable characters get focuses

			}
			10 = {
				modifier = { factor = 0 is_playable = no }
				set_character_flag = twl_quest_train
			}
			10 = {
				modifier = { factor = 0 is_playable = no }
				set_character_flag = twl_investigate_court
			}
			50 = { # AI bypasses missions at times
				modifier = {
					factor = 0
					ai = no
				}
				random_list = {
					70 = {
						modifier = {
							factor = 0.9
							society_influence >= 30

							factor = 0.9

							factor = 0.7

						modifier = {
							factor = 0.5
							society_influence >= 90
						}
						add_mission_succeed_influence_effect = yes
					}
					30 = {
						add_mission_fail_influence_effect = yes
					}
				}
			}

				modifier = {
					factor = 0
					ai = no
				}
