namespace = SWAMP

character_event = {
	id = SWAMP.1
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		OR = {
			trait = inbred
			trait = swamp_inbred
		}
	}

	immediate = {
		if = {
			limit = {
				religion = the_great_swamp
			}
			remove_trait = inbred
			add_trait = swamp_inbred
		}
		else = {
			remove_trait = swamp_inbred
			add_trait = inbred
		}
	}
}
