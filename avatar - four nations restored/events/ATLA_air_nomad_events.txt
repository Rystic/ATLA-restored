	id = NOMAD.15
	desc = NOMADEVTDESC15
	picture = GFX_evt_nomad_wander
	is_triggered_only = yes
	culture_group = air_nomad_culture
	trigger = {
		liege = {
			culture_group = air_nomad_culture
		}
		# ???
		NOT = {
			trait = avatar
		}
		NOT = {
			trait = fullyrealisedavatar
		}
		NOT = {
			trait = directdescendantavatar
		}
		NOT = {
			trait = descendantavatar
		}
	}
	option = {
		name = EVTOPTANOMADA.15
		remove_trait = airinitiate
		add_trait = airnomad
		if = {
			limit = {
				is_landed = no
			}
			liege = {
				# Ask Liege
				hidden_tooltip = {
					character_event = {
						id = NOMAD.401
					}
				}
			}
		}
		ai_chance = {
			factor = 10
			modifier = {
				factor = 0.5
				OR = {
					trait = content
					trait = shy
					trait = craven
				}
			}
		}
	}
	# REWORK #

		name = EVTOPTANOMADB.15
		remove_trait = airinitiate
		add_trait = airnomad
		if = {
			limit = {
				is_ruler = no
			}
			liege = {
				# Tell Liege you refuse
				hidden_tooltip = {
					character_event = {
						id = NOMAD.402
					}
				}
			}
		}
		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				NOR = {
					trait = content
					trait = shy
					trait = craven
				}
			}
		}
	}
}
