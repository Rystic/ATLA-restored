# On-action event: Right Home
character_event = {
	id = NOMAD.8
	hide_window = yes

	is_triggered_only = yes
	culture_group = air_nomad_culture
	ai = yes
	
	trigger = {
		is_ruler = no

		OR = {
			AND = {
				is_female = yes
				culture = northern_air_nomad
			}
			AND = {
				is_female = yes
				culture = southern_air_nomad
			}
			AND = {
				is_female = no
				culture = eastern_air_nomad
			}
			AND = {
				is_female = no
				culture = western_air_nomad
			}
			AND = {
				is_female = yes
				liege = {
					is_female = no
				}
			}
			AND = {
				is_female = no
				liege = {
					is_female = yes
				}
			}
		}
	}

	option = {
		name = OK
		if = {
			limit = {
				is_unimportant_trigger = yes
			}
			death = { death_reason = death_missing }
			break = yes
		}
		if = { # Female
			limit = {
				is_female = yes
			}
			random_list = {
				50 = {
					any_playable_ruler = {
						limit = {
							is_female = yes
							culture = eastern_air_nomad
						}
						ROOT = { 
							force_host = PREV 
							culture = eastern_air_nomad
						}
					}
				}
				50 = {
					any_playable_ruler = {
						limit = {
							is_female = yes
							culture = western_air_nomad
						}
						ROOT = { 
							force_host = PREV 
							culture = western_air_nomad
						}
					}
				}
			}
		}
		if = { # Male
			limit = {
				is_female = no
			}
			random_list = {
				50 = {
					any_playable_ruler = {
						limit = {

							force_host = PREV

						}
