character_event = { # Dai Li Spawn
	id = ATLAE.1000
	is_triggered_only = yes
	hide_window = yes

	trigger = {
		primary_title = {
			title = d_dai_li
		}
		NOT = { trait = headdaili }
	}

	option = {
		add_trait = headdaili
		if = {
			limit = {
				NOT = { intrigue = 12 }
			}
			change_intrigue = 12
		}
		any_courtier = {
			add_trait = daili
			limit = {
				prisoner = no
				is_ruler = no
				is_female = no
			}
		}
	}
}

character_event = { # Kyoshi Warriors Spawn
	id = ATLAE.1001
	is_triggered_only = yes
	hide_window = yes

	trigger = {
		primary_title = {
			title = d_kyoshi_warriors
		}
		is_female = yes
		any_courtier = {
			is_female = no
		}
	}

	option = {
		any_courtier = {
			if = {
				limit = {
					is_unimportant_trigger = yes
					is_female = no
				}
				death = { death_reason = death_missing }
				break = yes
			}
		}
	}
}

character_event = { # Dai Li take regency - ROOT
	id = ATLAE.1003
	is_triggered_only = yes
	hide_window = yes

	trigger = {
		primary_title = {
			title = d_dai_li
		}
		k_earth_kingdom = {
			holder_scope = {
				war = no
				has_regent = yes
				regent = {
					NOT = { has_landed_title = d_dai_li }
				}
			}
		}
	}

	option = {
		k_earth_kingdom = {
			holder_scope = {
				character_event = { id = ATLAE.1004 days = 9 }
			}
		}
	}
}

character_event = { # Dai Li take regency - EARTH KING
	id = ATLAE.1004
	desc = "EVTDESCORDER.1"
	is_triggered_only = yes
	picture = "GFX_evt_dai_li_2"

	trigger = {
		trait = earthkingpuppet
	}

	option = {
		name = EVTOPTORDERA.1
		regent = {
			remove_title = title_regent
		}
		FROM = {
			set_defacto_liege = ROOT
			give_minor_title = title_appointed_regent
			give_job_title = job_spymaster
		}
		create_character = {
				random_traits = yes
				dynasty = random
				religion = ROOT
				culture = ROOT
				age = 23
				trait = tough_soldier
				trait = daili
		}
		create_character = {
				random_traits = yes
				dynasty = random
				religion = ROOT
				culture = ROOT
				age = 27
				trait = tough_soldier
				trait = daili
		}
	}
}
