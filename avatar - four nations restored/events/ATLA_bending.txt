	desc = EVTDESCbending.1	#Look! I warmed the tea without fire!

		OR = {

			}

				mother_even_if_dead = {
					trait = firebender

			}

					trait = firebender
				}
			}
			mother_even_if_dead = {
				mother_even_if_dead = {
					trait = firebender
				}
			}
			mother_even_if_dead = {
				true_father_even_if_dead = {
					trait = firebender
				}
			}
			AND = {
				NOT = {
					has_game_rule = {
						name = bending_events
						value = rare
					}
				}
				culture_group = fire_nation_culture
			}
		}
	}
	option = {
		name = EVTOPTAbending.1		#Look! I warmed the water!

			character_event = {
				id = bending.2

		}
	}
