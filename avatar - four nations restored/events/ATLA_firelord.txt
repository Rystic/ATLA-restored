character_event = {
	id = firelord.1
	is_triggered_only = yes
	hide_window = yes

	trigger = {
		primary_title = {
			title = e_fire_nation
		}
		NOT = { trait = firelord }
		independent = yes
	}
	option = {
		name = EVTOPTfirelord.1
		if = {
			limit = {
				NOT = { check_variable = { which = zeal value = -100 } }
			}
			set_variable = { which = zeal value = 50 }
		}
		add_trait = firelord
		set_variable = { which = zeal value = 5 }
		character_event = { id = ATLAE.653 }
	}
}
