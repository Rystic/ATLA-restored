	employ_bender = {
		ai = no
		
		potential = {
			OR = {
				culture_group = earth_kingdom_cultures
				culture_group = fire_nation_cultures
				culture_group = water_tribe_cultures
				culture_group = air_nomad_cultures
			}
		}
		allow = {
			prestige = 20
			scaled_wealth = 0.3
			NOT = {
				has_character_modifier = in_seclusion
			}
		}
		effect = {
			prestige = -20
			scaled_wealth = -0.3
			create_random_soldier = {
				random_traits = yes
				female = 50 #50/50 to be a woman, bending doesn't discriminate
				dynasty = actually_culture
				culture = ROOT
				religion = ROOT
			}
			trigger_switch = {
				on_trigger = culture_group
				earth_kingdom_cultures = { new_character = { trait = earthbender trait = competent_bender } }
				fire_nation_cultures = { new_character = { trait = firebender trait = competent_bender } }
				water_tribe_cultures = { new_character = { trait = waterbender trait = competent_bender } }
				air_nomad_cultures = { new_character = { trait = airbender trait = competent_bender } }
			}
			new_character = {
				remove_trait = weak
				remove_trait = slow
				remove_trait = imbecile
				remove_trait = dull
				remove_trait = craven	
				if = {
					limit = {
						trait = misguided_warrior
					}
					remove_trait = misguided_warrior
					add_trait = tough_soldier
				}
				random = {
					chance = 20
					upgrade_bending_effect = yes
				}
				save_event_target_as = invited_character
			}
			character_event = { id = NE.1 }
		}
	}
